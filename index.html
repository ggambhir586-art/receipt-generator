       function downloadPDF() {

           const element = document.getElementById('receipt-content');

           const txnId = document.getElementById('txn-id').innerText;

         

           // Hide borders/placeholder text for cleaner look

           const inputs = element.querySelectorAll('input');

           inputs.forEach(i => i.style.borderBottom = 'none');

           const opt = {

               margin: 0,

               filename: `University_Receipt_${txnId}.pdf`,

               image: { type: 'jpeg', quality: 0.98 },

               html2canvas: { scale: 2, useCORS: true, scrollY: 0 },

               jsPDF: { unit: 'mm', format: 'a4', orientation: 'portrait' }

           };

           html2pdf().set(opt).from(element).save().then(() => {

               // Restore styling

               inputs.forEach(i => i.style.borderBottom = '1px dotted #d1d5db');

           });

       }

           document.getElementById('words-display').innerText = convertNumberToWords(total) + " Only";

       }

       function convertNumberToWords(amount) {

           if (amount === 0) return "Zero Taka";

           const ones = ['', 'One', 'Two', 'Three', 'Four', 'Five', 'Six', 'Seven', 'Eight', 'Nine', 'Ten', 'Eleven', 'Twelve', 'Thirteen', 'Fourteen', 'Fifteen', 'Sixteen', 'Seventeen', 'Eighteen', 'Nineteen'];

           const tens = ['', '', 'Twenty', 'Thirty', 'Forty', 'Fifty', 'Sixty', 'Seventy', 'Eighty', 'Ninety'];

           function convertGroup(n) {

               if (n < 20) return ones[n];

               const digit = n % 10;

               return tens[Math.floor(n / 10)] + (digit ? " " + ones[digit] : "");

           }

           function convertHundreds(n) {

               if (n > 99) {

                   return ones[Math.floor(n / 100)] + " Hundred" + (n % 100 ? " " + convertGroup(n % 100) : "");

               } else {

                   return convertGroup(n);

               }

           }

           let intPart = Math.floor(amount);

           let words = "";

           if (intPart >= 10000000) {

               words += convertHundreds(Math.floor(intPart / 10000000)) + " Crore ";

               intPart %= 10000000;

           }

           if (intPart >= 100000) {

               words += convertHundreds(Math.floor(intPart / 100000)) + " Lakh ";

               intPart %= 100000;

           }

           if (intPart >= 1000) {

               words += convertHundreds(Math.floor(intPart / 1000)) + " Thousand ";

               intPart %= 1000;

           }

           if (intPart > 0) {

               words += convertHundreds(intPart);

           }

           return words.trim() + " Taka";

       }

       function handleLogoUpload(input) {

           if (input.files && input.files[0]) {

               var reader = new FileReader();

               reader.onload = function (e) {

                   // Header Logo Update

                   const img = document.querySelector('.custom-logo-img');

                   const defaultContent = document.querySelector('.default-logo-content');

                 

                   img.src = e.target.result;

                   img.classList.remove('hidden');

                   defaultContent.classList.add('hidden');

                   // Watermark Update

                   const defaultWatermark = document.getElementById('default-watermark');

                   const customWatermark = document.getElementById('custom-watermark');

                 

                   customWatermark.src = e.target.result;

                   customWatermark.classList.remove('hidden');

                   defaultWatermark.classList.add('hidden');

               };

               reader.readAsDataURL(input.files[0]);

           }

       }
